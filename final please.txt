@app.post("/v2/generate_context", response_model=Dict)
async def generate_reasoning_context(request: Dict[str, str]) -> Dict:
    """
    Generate reasoning context using dual retrieval from TCL and Super-Graph.
    
    Args:
        request: Dict with user_input and optional user_id
        
    Returns:
        Dict with reasoning context and deliberative prompt
    """
    request_counter.labels(endpoint="generate_context").inc()
    
    with reasoning_duration.time():
        try:
            user_input = request.get("user_input", "")
            user_id = request.get("user_id")
            
            if not user_input:
                raise HTTPException(status_code=400, detail="user_input is required")
            
            # Generate reasoning context
            context = await reasoning_engine.generate_response_context(user_input, user_id)
            
            # Build deliberative prompt
            deliberative_prompt = reasoning_engine.build_deliberative_prompt(user_input, context)
            
            log_info("Reasoning context generated", 
                    user_id=user_id,
                    therapeutic_components=len(context.therapeutic_components),
                    factual_knowledge=len(context.factual_knowledge))
            
            return {
                "user_input": user_input,
                "context": context.dict(),
                "deliberative_prompt": deliberative_prompt,
                "generated_at": pd.Timestamp.now().isoformat()
            }
            
        except HTTPException:
            raise
        except Exception as e:
            logger.error(f"Context generation failed: {e}")
            raise HTTPException(status_code=500, detail="Context generation failed")

@app.post("/v2/validate_response", response_model=Dict)
async def validate_ai_response(request: Dict[str, any]) -> Dict:
    """
    Validate AI response for safety and factual accuracy.
    
    Args:
        request: Dict with draft_response and context
        
    Returns:
        Dict with validation results and safety status
    """
    request_counter.labels(endpoint="validate_response").inc()
    
    try:
        draft_response = request.get("draft_response", "")
        context = request.get("context", {})
        
        if not draft_response:
            raise HTTPException(status_code=400, detail="draft_response is required")
        
        # Run safety validation
        safety_result = safety_governor.validate_response_safety(draft_response, context)
        
        log_info("Response validated", 
                is_safe=safety_result["is_safe"],
                contradictions=len(safety_result["contradictions"]),
                warnings=len(safety_result["warnings"]))
        
        return {
            "draft_response": draft_response,
            "safety_validation": safety_result,
            "validated_at": pd.Timestamp.now().isoformat()
        }
        
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Response validation failed: {e}")
        raise HTTPException(status_code=500, detail="Response validation failed")

"""
TPN-Prime 5.0 Reasoning Engine with Dual Retrieval
Replaces simple pathway generation with context-aware reasoning.
"""
import logging
from typing import Dict, List, Optional
from jinja2 import Environment, FileSystemLoader
import json

from shared.database import db
from shared.knowledge_service import knowledge_service
from shared.models import TherapeuticComponent, ReasoningContext
from shared.logging_config import log_info

logger = logging.getLogger(__name__)

class ReasoningEngine:
    """
    Advanced reasoning engine that combines therapeutic components (TCL) 
    with external knowledge (Super-Graph) for contextual response generation.
    """
    
    def __init__(self):
        # Load Jinja2 templates for prompt engineering
        self.jinja_env = Environment(loader=FileSystemLoader('pdm/prompts'))
        
        # Initialize therapeutic component cache
        self.tcl_cache = {}
        
    async def generate_response_context(self, user_input: str, user_id: str = None) -> ReasoningContext:
        """
        Performs Dual Retrieval to build context for the LLM.
        
        Args:
            user_input: User's input text
            user_id: Optional user ID for personalization
            
        Returns:
            ReasoningContext with therapeutic and factual knowledge
        """
        try:
            # Query for relevant internal therapeutic skills (TCL)
            therapeutic_components = await self.get_relevant_pathways(user_input)
            
            # Query for external factual knowledge (Super-Graph)
            factual_knowledge = []
            if await knowledge_service.detect_factual_query(user_input):
                factual_knowledge = await knowledge_service.query_super_graph(user_input)
            
            # Get user profile summary if available
            user_profile_summary = {}
            if user_id:
                user_profile_summary = await self._get_user_profile_summary(user_id)
            
            context = ReasoningContext(
                therapeutic_components=therapeutic_components,
                factual_knowledge=factual_knowledge,
                user_profile_summary=user_profile_summary
            )
            
            log_info("Reasoning context generated", 
                    user_id=user_id,
                    therapeutic_count=len(therapeutic_components),
                    factual_count=len(factual_knowledge))
            
            return context
            
        except Exception as e:
            logger.error(f"Context generation failed: {e}")
            # Return empty context on failure
            return ReasoningContext(
                therapeutic_components=[],
                factual_knowledge=[],
                user_profile_summary={}
            )
    
    def build_deliberative_prompt(self, user_input: str, context: ReasoningContext) -> str:
        """
        Builds the final prompt for the LLM using a Jinja2 template.
        
        Args:
            user_input: User's original input
            context: Reasoning context with components and knowledge
            
        Returns:
            Formatted prompt string for LLM
        """
        try:
            template = self.jinja_env.get_template('deliberative_prompt.jinja2')
            
            prompt = template.render(
                user_input=user_input,
                therapeutic_components=context.therapeutic_components,
                factual_knowledge=context.factual_knowledge,
                user_profile=context.user_profile_summary
            )
            
            return prompt
            
        except Exception as e:
            logger.error(f"Prompt building failed: {e}")
            # Fallback to simple prompt
            return f"""
            User Input: {user_input}
            
            Please provide a helpful, therapeutic response based on evidence-based practices.
            Ensure accuracy and avoid making unsupported claims.
            """
    
    async def get_relevant_pathways(self, text: str) -> List[TherapeuticComponent]:
        """
        Query therapeutic components from the TCL based on user input.
        
        Args:
            text: User input text to match against
            
        Returns:
            List of relevant therapeutic components
        """
        try:
            # Search therapeutic components using text similarity
            results = await db.fetch_all(
                """
                SELECT component_id, pillar, theory, dora_category, 
                       skill_definition, execution_process, global_efficacy
                FROM therapeutic_components
                WHERE to_tsvector('english', 
                    pillar || ' ' || theory || ' ' || dora_category || ' ' || 
                    (skill_definition->>'name') || ' ' || (skill_definition->>'description')
                ) @@ plainto_tsquery('english', $1)
                ORDER BY global_efficacy DESC
                LIMIT 5
                """,
                text
            )
            
            components = []
            for row in results:
                component = TherapeuticComponent(
                    component_id=row["component_id"],
                    pillar=row["pillar"],
                    theory=row["theory"],
                    dora_category=row["dora_category"],
                    skill_definition=row["skill_definition"],
                    execution_process=row["execution_process"],
                    global_efficacy=row["global_efficacy"]
                )
                components.append(component)
            
            # If no matches, return default components
            if not components:
                components = await self._get_default_components()
            
            return components
            
        except Exception as e:
            logger.error(f"Therapeutic component retrieval failed: {e}")
            return await self._get_default_components()
    
    async def _get_user_profile_summary(self, user_id: str) -> Dict:
        """Get summarized user profile for personalization."""
        try:
            profile = await db.fetch_one(
                "SELECT assessments, cluster_id FROM user_profiles WHERE user_id = $1",
                user_id
            )
            
            if profile:
                return {
                    "cluster_id": profile["cluster_id"],
                    "assessment_count": len(profile["assessments"]) if profile["assessments"] else 0,
                    "has_profile": True
                }
            
            return {"has_profile": False}
            
        except Exception as e:
            logger.error(f"User profile summary failed for {user_id}: {e}")
            return {"has_profile": False}class TestFactsEndpoint:

# TPN-Prime 4.0 MVP - Modular Therapeutic Discovery Engine

## Overview

The TPN-Prime 4.0 MVP is a modular, extensible therapeutic AI system designed as a supervised discovery engine for personalized mental health interventions. It delivers a proof of concept (PoC) within a 2-month timeline, validating novel therapeutic pathways via an N-of-1 trial with 50 participants.

## Architecture

The system comprises six independently deployable modules:

- **UIM** (User Interaction Module) - Handles onboarding and session management

- **SRM** (Signal Reconciliation Module) - Reconciles real-world evidence with clinical standards
- **EFGM** (External Factual Graph Module) - Provides factual grounding via ADHD knowledge graph
- **MMM** (MLOps and Monitoring Module) - Manages deployment and research insights

# RDF Namespaces
ADHD = Namespace("http://example.org/adhd/")
THERAPY = Namespace("http://example.org/therapy/")
RDFS = Namespace("http://www.w3.org/2000/01/rdf-schema#")

class KnowledgeGraphManager:
    """Manages ADHD Knowledge Graph operations and querying."""
    
    def __init__(self):
        self.graph = Graph()
        self.query_cache = {}  # Simple query result cache
        self.max_cache_size = 100
        
        # Bind namespaces
        self.graph.bind("adhd", ADHD)
        self.graph.bind("therapy", THERAPY)
        self.graph.bind("rdfs", RDFS)
        
        # Common SPARQL queries
        self.queries = {
            "treatments_for_symptom": """
                SELECT ?treatment ?label ?effectiveness WHERE {
                    ?treatment therapy:treats ?symptom .
                    ?treatment rdfs:label ?label .
                    OPTIONAL { ?treatment therapy:effectiveness ?effectiveness }
                    FILTER(CONTAINS(LCASE(STR(?symptom)), LCASE("%s")))
                }
            """,
            "symptoms_of_condition": """
                SELECT ?symptom ?label ?severity WHERE {
                    ?symptom rdfs:subClassOf ?condition .
                    ?symptom rdfs:label ?label .
                    OPTIONAL { ?symptom adhd:severity ?severity }
                    FILTER(CONTAINS(LCASE(STR(?condition)), LCASE("%s")))
                }
            """,
            "related_interventions": """
                SELECT ?intervention ?label ?type WHERE {
                    ?intervention a ?type .
                    ?intervention rdfs:label ?label .
                    FILTER(?type = therapy:Therapy || ?type = therapy:Lifestyle || ?type = adhd:Medication)
                }
            """,
            "comorbidities": """
                SELECT ?comorbidity ?label WHERE {
                    ?comorbidity adhd:coOccursWith adhd:ADHD .
                    ?comorbidity rdfs:label ?label .
                }
            """
        }
    
    async def initialize_graph(self) -> None:
        """Initialize knowledge graph from database and file sources."""
        try:
            # Load from Turtle file
            await self._load_from_file()
            
            # Load from database
            await self._load_from_database()
            
            log_info("Knowledge graph initialized", 
                    triples_count=len(self.graph),
                    namespaces=len(list(self.graph.namespaces())))
            
        except Exception as e:
            logger.error(f"Failed to initialize knowledge graph: {e}")
            raise
    
    async def _load_from_file(self) -> None:
        """Load knowledge graph from TTL file."""
        try:
            self.graph.parse("data/adhd_kg_subset.ttl", format="turtle")
            log_info("Loaded graph from TTL file", triples_from_file=len(self.graph))
        except Exception as e:
            logger.warning(f"Failed to load from TTL file: {e}")
            # Create minimal graph if file loading fails
            await self._create_minimal_graph()
    
    async def _load_from_database(self) -> None:
        """Load additional triples from database."""
        try:
            triples = await db.fetch_all("SELECT entity, relation, object FROM adhd_kg")
            
            for triple in triples:
                subject = URIRef(triple["entity"])
                predicate = URIRef(triple["relation"])
                obj = URIRef(triple["object"]) if triple["object"].startswith("http") else Literal(triple["object"])
                
                self.graph.add((subject, predicate, obj))
            
            log_info("Loaded graph from database", triples_from_db=len(triples))
            
        except Exception as e:
            logger.warning(f"Failed to load from database: {e}")
    
    async def _create_minimal_graph(self) -> None:
        """Create minimal knowledge graph for fallback."""
        # Add basic ADHD concepts
        self.graph.add((ADHD.Inattention, RDFS.subClassOf, ADHD.ADHD))
        self.graph.add((ADHD.Inattention, RDFS.label, Literal("Inattention")))
        self.graph.add((ADHD.Hyperactivity, RDFS.subClassOf, ADHD.ADHD))
        self.graph.add((ADHD.Hyperactivity, RDFS.label, Literal("Hyperactivity")))
        
        # Add basic treatments
        self.graph.add((THERAPY.CBT, THERAPY.treats, ADHD.Inattention))
        self.graph.add((THERAPY.CBT, RDFS.label, Literal("Cognitive Behavioral Therapy")))
        self.graph.add((THERAPY.Mindfulness, THERAPY.treats, ADHD.Hyperactivity))
        self.graph.add((THERAPY.Mindfulness, RDFS.label, Literal("Mindfulness Meditation")))
        
        log_info("Created minimal fallback graph")
    
    async def etl_from_source(self, source_data: List[Dict]) -> int:
        """
        ETL pipeline to ingest new knowledge graph data.
        
        Args:
            source_data: List of triples as dictionaries.
            
        Returns:
            Number of triples processed.
        """
        try:
            processed_count = 0
            
            for item in source_data:
                subject = URIRef(item.get("subject", item.get("entity")))
                predicate = URIRef(item.get("predicate", item.get("relation")))
                obj_value = item.get("object", item.get("obj"))
                
                # Determine if object is URI or literal
                if isinstance(obj_value, str) and obj_value.startswith("http"):
                    obj = URIRef(obj_value)
                else:
                    obj = Literal(obj_value)
                
                # Add to graph
                self.graph.add((subject, predicate, obj))
                
                # Store in database
                await db.execute_query(
                    "INSERT INTO adhd_kg (entity, relation, object) VALUES ($1, $2, $3) ON CONFLICT DO NOTHING",
                    str(subject), str(predicate), str(obj)
                )
                
                processed_count += 1
            
            # Clear cache after ETL
            self.query_cache.clear()
            
            log_info("ETL pipeline completed", processed_triples=processed_count)
            return processed_count
            
        except Exception as e:
            logger.error(f"ETL pipeline failed: {e}")
            raise
    
    def _cache_query_result(self, query_key: str, result: List[Dict]) -> None:
        """Cache query result with LRU-style eviction."""
        if len(self.query_cache) >= self.max_cache_size:
            # Remove oldest entry (simple FIFO for MVP)
            oldest_key = next(iter(self.query_cache))
            del self.query_cache[oldest_key]
        
        self.query_cache[query_key] = result
    
    async def query_treatments_for_symptom(self, symptom: str) -> List[Dict]:
        """
        Query treatments for a specific symptom.
        
        Args:
            symptom: Symptom name to search for.
            
        Returns:
            List of treatment dictionaries.
        """
        query_key = f"treatments_{symptom.lower()}"
        
        # Check cache
        if query_key in self.query_cache:
            return self.query_cache[query_key]
        
        try:
            query = self.queries["treatments_for_symptom"] % symptom
            results = self.graph.query(query)
            
            treatments = []
            for row in results:
                treatment_data = {
                    "treatment": str(row.treatment),
                    "label": str(row.label) if row.label else str(row.treatment),
                    "effectiveness": str(row.effectiveness) if row.effectiveness else "unknown"
                }
                treatments.append(treatment_data)
            
            # Cache result
            self._cache_query_result(query_key, treatments)
            
            return treatments
            
        except Exception as e:
            logger.error(f"Treatment query failed for symptom {symptom}: {e}")
            return []
    
    async def query_symptoms_of_condition(self, condition: str) -> List[Dict]:
        """
        Query symptoms of a specific condition.
        
        Args:
            condition: Condition name to search for.
            
        Returns:
            List of symptom dictionaries.
        """
        query_key = f"symptoms_{condition.lower()}"
        
        # Check cache
        if query_key in self.query_cache:
            return self.query_cache[query_key]
        
        try:
            query = self.queries["symptoms_of_condition"] % condition
            results = self.graph.query(query)
            
            symptoms = []
            for row in results:
                symptom_data = {
                    "symptom": str(row.symptom),
                    "label": str(row.label) if row.label else str(row.symptom),
                    "severity": str(row.severity) if row.severity else "unknown"
                }
                symptoms.append(symptom_data)
            
            # Cache result
            self._cache_query_result(query_key, symptoms)
            
            return symptoms
            
        except Exception as e:
            logger.error(f"Symptom query failed for condition {condition}: {e}")
            return []
    
    async def query_related_interventions(self) -> List[Dict]:
        """
        Query all available interventions.
        
        Returns:
            List of intervention dictionaries.
        """
        query_key = "all_interventions"
        
        # Check cache
        if query_key in self.query_cache:
            return self.query_cache[query_key]
        
        try:
            query = self.queries["related_interventions"]
            results = self.graph.query(query)
            
            interventions = []
            for row in results:
                intervention_data = {
                    "intervention": str(row.intervention),
                    "label": str(row.label) if row.label else str(row.intervention),
                    "type": str(row.type).split("/")[-1] if row.type else "unknown"
                }
                interventions.append(intervention_data)
            
            # Cache result
            self._cache_query_result(query_key, interventions)
            
            return interventions
            
        except Exception as e:
            logger.error(f"Intervention query failed: {e}")
            return []
    
    async def query_comorbidities(self) -> List[Dict]:
        """
        Query ADHD comorbidities.
        
        Returns:
            List of comorbidity dictionaries.
        """
        query_key = "adhd_comorbidities"
        
        # Check cache
        if query_key in self.query_cache:
            return self.query_cache[query_key]
        
        try:
            query = self.queries["comorbidities"]
            results = self.graph.query(query)
            
            comorbidities = []
            for row in results:
                comorbidity_data = {
                    "comorbidity": str(row.comorbidity),
                    "label": str(row.label) if row.label else str(row.comorbidity)
                }
                comorbidities.append(comorbidity_data)
            
            # Cache result
            self._cache_query_result(query_key, comorbidities)
            
            return comorbidities
            
        except Exception as e:
            logger.error(f"Comorbidity query failed: {e}")
            return []
    
    async def custom_sparql_query(self, sparql_query: str) -> List[Dict]:
        """
        Execute custom SPARQL query.
        
        Args:
            sparql_query: SPARQL query string.
            
        Returns:
            List of result dictionaries.
        """
        try:
            results = self.graph.query(sparql_query)
            
            formatted_results = []
            for row in results:
                row_data = {}
                for var in results.vars:
                    value = getattr(row, str(var), None)
                    row_data[str(var)] = str(value) if value else None
                formatted_results.append(row_data)
            
            return formatted_results
            
        except Exception as e:
            logger.error(f"Custom SPARQL query failed: {e}")
            raise HTTPException(status_code=400, detail=f"SPARQL query failed: {str(e)}")

@app.get("/facts/{topic}", response_model=Dict)
async def get_facts(
    topic: str,
    limit: int = Query(default=10, ge=1, le=100),
    query_type: str = Query(default="auto", regex="^(auto|treatments|symptoms|interventions|comorbidities)$")
) -> Dict:
    """
    Query factual information about a topic from the knowledge graph.
    
    Args:
        topic: Topic to query (e.g., 'adhd', 'inattention').
        limit: Maximum number of facts to return.
        query_type: Type of query (auto, treatments, symptoms, interventions, comorbidities).
        
    Returns:
        Dict with facts and metadata.
    """
    request_counter.labels(endpoint="facts").inc()
    
    with query_duration.time():
        try:
            facts = []
            
            if query_type == "auto" or query_type == "treatments":
                treatments = await kg_manager.query_treatments_for_symptom(topic)
                facts.extend([{**t, "fact_type": "treatment"} for t in treatments])
            
            if query_type == "auto" or query_type == "symptoms":
                symptoms = await kg_manager.query_symptoms_of_condition(topic)
                facts.extend([{**s, "fact_type": "symptom"} for s in symptoms])
            
            if query_type == "auto" or query_type == "interventions":
                interventions = await kg_manager.query_related_interventions()
                facts.extend([{**i, "fact_type": "intervention"} for i in interventions])
            
            if query_type == "auto" or query_type == "comorbidities":
                comorbidities = await kg_manager.query_comorbidities()
                facts.extend([{**c, "fact_type": "comorbidity"} for c in comorbidities])
            
            # Apply limit
            facts = facts[:limit]
            
            log_info("Facts retrieved", topic=topic, query_type=query_type, 
                    fact_count=len(facts), cache_size=len(kg_manager.query_cache))
            
            return {
                "topic": topic,
                "query_type": query_type,
                "facts": facts,
                "count": len(facts),
                "graph_size": len(kg_manager.graph),
                "cache_hit_ratio": len(kg_manager.query_cache) / kg_manager.max_cache_size
            }
            
        except Exception as e:
            logger.error(f"Fact retrieval failed for topic {topic}: {e}")
            raise HTTPException(status_code=500, detail="Fact retrieval failed")

@app.post("/sparql", response_model=Dict)
async def execute_sparql_query(query_data: Dict[str, str]) -> Dict:
    """
    Execute custom SPARQL query against the knowledge graph.
    
    Args:
        query_data: Dict with 'query' key containing SPARQL query string.
        
    Returns:
        Dict with query results.
        
    Raises:
        HTTPException: If query is invalid or execution fails.
    """
    request_counter.labels(endpoint="sparql").inc()
    
    try:
        sparql_query = query_data.get("query", "")
        if not sparql_query:
            raise HTTPException(status_code=400, detail="No SPARQL query provided")
        
        # Basic query validation (prevent dangerous operations)
        forbidden_keywords = ["DELETE", "INSERT", "DROP", "CLEAR", "LOAD"]
        query_upper = sparql_query.upper()
        
        for keyword in forbidden_keywords:
            if keyword in query_upper:
                raise HTTPException(
                    status_code=400,
                    detail=f"Query contains forbidden operation: {keyword}"
                )
        
        with query_duration.time():
            results = await kg_manager.custom_sparql_query(sparql_query)
        
        log_info("SPARQL query executed", result_count=len(results))
        
        return {
            "query": sparql_query,
            "results": results,
            "count": len(results)
        }
        
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"SPARQL query execution failed: {e}")
        raise HTTPException(status_code=500, detail="SPARQL query execution failed")

@app.get("/graph/stats", response_model=Dict)

    """
    Get knowledge graph statistics and metadata.
    
    Returns:
        Dict with graph statistics.
    """
    request_counter.labels(endpoint="graph_stats").inc()
    
    try:
        # Count different types of entities
        entity_types = {}
        predicate_counts = {}
        
        for subject, predicate, obj in kg_manager.graph:
            # Count predicates
            pred_str = str(predicate).split("/")[-1]
            predicate_counts[pred_str] = predicate_counts.get(pred_str, 0) + 1
            
            # Count entity types (simplified)
            if "adhd" in str(subject).lower():
                entity_types["adhd_entities"] = entity_types.get("adhd_entities", 0) + 1
            elif "therapy" in str(subject).lower():
                entity_types["therapy_entities"] = entity_types.get("therapy_entities", 0) + 1
        
        return {
            "total_triples": len(kg_manager.graph),
            "total_namespaces": len(list(kg_manager.graph.namespaces())),
            "entity_type_distribution": entity_types,
            "predicate_distribution": predicate_counts,
            "cache_statistics": {
                "cached_queries": len(kg_manager.query_cache),
                "cache_max_size": kg_manager.max_cache_size,
                "cache_usage": len(kg_manager.query_cache) / kg_manager.max_cache_size
            }
        }
        
    except Exception as e:

        raise HTTPException(status_code=500, detail="Statistics retrieval failed")

async def collect_system_metrics() -> None:
    """Background task to collect system metrics."""
    while True:
        try:
            # Update active users gauge
            user_count = await db.fetch_one(
                "SELECT COUNT(DISTINCT user_id) as count FROM user_profiles"
            )
            if user_count:
                active_users_gauge.set(user_count["count"])
            
            # Update pathways gauge
            pathway_count = await db.fetch_one(
                "SELECT COUNT(*) as count FROM pathways"
            )
            if pathway_count:
                pathways_gauge.set(pathway_count["count"])
            
            await asyncio.sleep(60)  # Update every minute
            
        except Exception as e:
            logger.error(f"Metrics collection failed: {e}")
            await asyncio.sleep(60)

@app.get("/stats/overview", response_model=Dict)
async def get_overview_statistics() -> Dict:
    """
    Get high-level system overview statistics.
    
    Returns:
        Dict with overview statistics.
    """
    request_counter.labels(endpoint="overview_stats").inc()
    
    try:
        # User statistics
        user_stats = await db.fetch_one(
            """SELECT COUNT(*) as total_users,
                      COUNT(DISTINCT cluster_id) as clusters,
                      AVG(array_length(profile_vector, 1)) as avg_vector_length
               FROM user_profiles"""
        )
        
        # Pathway statistics
        pathway_stats = await db.fetch_one(
            """SELECT COUNT(*) as total_pathways,
                      COUNT(CASE WHEN status = 'VALIDATED' THEN 1 END) as validated_pathways,
                      AVG(meta_score) as avg_meta_score,
                      AVG(confidence) as avg_confidence
               FROM pathways"""
        )
        
        # Session statistics
        session_stats = await db.fetch_one(
            """SELECT COUNT(*) as total_sessions,
                      COUNT(DISTINCT user_id) as active_users,
                      AVG(phq9_score) as avg_phq9,
                      AVG(sentiment_score) as avg_sentiment
               FROM user_sessions"""
        )
        
        # System health check
        health_checks = {}
        services = ["UIM", "PDM", "TVM", "SRM", "EFGM"]
        urls = [config.UIM_URL, config.PDM_URL, config.TVM_URL, config.SRM_URL, config.EFGM_URL]
        
        for service, url in zip(services, urls):
            try:
                async with httpx.AsyncClient(timeout=5.0) as client:
                    response = await client.get(f"{url}/health")
                    health_checks[service] = "healthy" if response.status_code == 200 else "unhealthy"
            except Exception:
                health_checks[service] = "unhealthy"
        
        return {
            "user_statistics": {
                "total_users": user_stats["total_users"] or 0,
                "clusters": user_stats["clusters"] or 0,
                "avg_vector_length": float(user_stats["avg_vector_length"] or 0)
            },
            "pathway_statistics": {
                "total_pathways": pathway_stats["total_pathways"] or 0,
                "validated_pathways": pathway_stats["validated_pathways"] or 0,
                "avg_meta_score": float(pathway_stats["avg_meta_score"] or 0),
                "avg_confidence": float(pathway_stats["avg_confidence"] or 0)
            },
            "session_statistics": {
                "total_sessions": session_stats["total_sessions"] or 0,
                "active_users": session_stats["active_users"] or 0,
                "avg_phq9": float(session_stats["avg_phq9"] or 0),
                "avg_sentiment": float(session_stats["avg_sentiment"] or 0)
            },
            "system_health": health_checks,
            "generated_at": pd.Timestamp.now().isoformat()
        }
        
    except Exception as e:
        logger.error(f"Failed to retrieve overview statistics: {e}")
        raise HTTPException(status_code=500, detail="Statistics retrieval failed")

"""

import asyncpg
import json
import logging
from typing import Any, Dict, List, Optional
from tenacity import retry, stop_after_attempt, wait_fixed
from .config import config

logger = logging.getLogger(__name__)

class DatabaseManager:
    """Manages database connections and operations with retry logic."""
    
    def __init__(self, db_url: str = None):
        self.db_url = db_url or config.DB_URL
        
    @retry(stop=stop_after_attempt(3), wait=wait_fixed(1))
    async def get_connection(self) -> asyncpg.Connection:
        """Get database connection with retry logic."""
        try:
            conn = await asyncpg.connect(self.db_url)
            return conn
        except Exception as e:
            logger.error(f"Database connection failed: {e}")
            raise
    
    @retry(stop=stop_after_attempt(3), wait=wait_fixed(1))
    async def execute_query(self, query: str, *args) -> None:
        """Execute a query with retry logic."""
        conn = await self.get_connection()
        try:
            await conn.execute(query, *args)
        finally:
            await conn.close()
    
    @retry(stop=stop_after_attempt(3), wait=wait_fixed(1))
    async def fetch_one(self, query: str, *args) -> Optional[Dict[str, Any]]:
        """Fetch one record with retry logic."""
        conn = await self.get_connection()
        try:
            result = await conn.fetchrow(query, *args)
            return dict(result) if result else None
        finally:
            await conn.close()
    
    @retry(stop=stop_after_attempt(3), wait=wait_fixed(1))
    async def fetch_all(self, query: str, *args) -> List[Dict[str, Any]]:
        """Fetch all records with retry logic."""
        conn = await self.get_connection()
        try:
            results = await conn.fetch(query, *args)
            return [dict(row) for row in results]
        finally:
            await conn.close()

db = DatabaseManager()

"""
Structured JSON logging configuration
"""
import logging
import json
import sys
from datetime import datetime
from typing import Any, Dict

class JSONFormatter(logging.Formatter):
    """Custom JSON formatter for structured logging."""
    
    def format(self, record: logging.LogRecord) -> str:
        log_data = {
            "timestamp": datetime.utcnow().isoformat(),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
            "module": record.module,
            "function": record.funcName,
            "line": record.lineno
        }
        
        if hasattr(record, 'user_id'):
            log_data['user_id'] = record.user_id
        if hasattr(record, 'pathway_id'):
            log_data['pathway_id'] = record.pathway_id
        if hasattr(record, 'cluster_id'):
            log_data['cluster_id'] = record.cluster_id
            
        return json.dumps(log_data)

def setup_logging(level: str = "INFO") -> None:
    """Setup structured JSON logging."""
    logger = logging.getLogger()
    logger.setLevel(getattr(logging, level.upper()))
    
    handler = logging.StreamHandler(sys.stdout)
    handler.setFormatter(JSONFormatter())
    
    logger.addHandler(handler)

def log_info(message: str, **kwargs) -> None:
    """Log info message with additional context."""
    logger = logging.getLogger()
    extra = {k: v for k, v in kwargs.items()}
    logger.info(message, extra=extra)

"""

"""
import logging
import hashlib
from typing import List, Dict, Optional, Any
import re
from shared.database import db
from shared.models import ExternalKnowledge

logger = logging.getLogger(__name__)

class KnowledgeService:
    """Unified service for querying the Super-Graph (external + internal knowledge)."""
    
    def __init__(self):
        # Patterns to detect factual queries
        self.factual_query_patterns = [
            r'\b(?:what is|what are|define|explain|tell me about)\b',
            r'\b(?:how does|how do|mechanism|process)\b',
            r'\b(?:side effects?|contraindications?|interactions?)\b',
            r'\b(?:effectiveness|efficacy|studies|research)\b',
            r'\b(?:dosage|dose|administration)\b'
        ]
    
    async def detect_factual_query(self, user_input: str) -> bool:
        """
        Detect if user input is requesting factual information.
        
        Args:
            user_input: User's text input
            
        Returns:
            Boolean indicating if this appears to be a factual query
        """
        input_lower = user_input.lower()
        
        for pattern in self.factual_query_patterns:
            if re.search(pattern, input_lower):
                return True
        
        # Check for medication/drug names (simplified)
        drug_indicators = ['medication', 'drug', 'pill', 'prescription', 'antidepressant']
        if any(indicator in input_lower for indicator in drug_indicators):
            return True
            
        return False
    
    async def query_super_graph(self, query: str, limit: int = 10) -> List[ExternalKnowledge]:
        """
        Query the Super-Graph for relevant external knowledge.
        
        Args:
            query: Search query
            limit: Maximum number of results
            
        Returns:
            List of relevant external knowledge facts
        """
        try:
            # Extract key terms from query for search
            search_terms = self._extract_search_terms(query)
            
            if not search_terms:
                return []
            
            # Search external knowledge using full-text search
            results = await db.fetch_all(
                """
                SELECT entity_id, source_kg, entity_type, entity_name, 
                       relationship_type, related_entity_name, source_link, metadata
                FROM external_knowledge
                WHERE to_tsvector('english', entity_name || ' ' || COALESCE(related_entity_name, '')) 
                      @@ plainto_tsquery('english', $1)
                ORDER BY ts_rank(to_tsvector('english', entity_name), plainto_tsquery('english', $1)) DESC
                LIMIT $2
                """,
                ' '.join(search_terms), limit
            )
            
            knowledge_facts = []
            for row in results:
                knowledge_facts.append(ExternalKnowledge(
                    entity_id=row["entity_id"],
                    source_kg=row["source_kg"],
                    entity_type=row["entity_type"],
                    entity_name=row["entity_name"],
                    relationship_type=row["relationship_type"],
                    related_entity_name=row["related_entity_name"],
                    source_link=row["source_link"],
                    metadata=row["metadata"] or {}
                ))
            
            return knowledge_facts
            
        except Exception as e:
            logger.error(f"Super-Graph query failed: {e}")
            return []
    
    async def ingest_external_knowledge(self, knowledge_facts: List[Dict[str, Any]]) -> int:
        """
        Ingest external knowledge facts into the Super-Graph.
        
        Args:
            knowledge_facts: List of knowledge fact dictionaries
            
        Returns:
            Number of facts successfully ingested
        """
        try:
            ingested_count = 0
            
            for fact in knowledge_facts:
                # Generate fingerprint for deduplication
                fingerprint = self._generate_fingerprint(fact)
                
                # Check if already exists
                existing = await db.fetch_one(
                    "SELECT id FROM external_knowledge WHERE fingerprint = $1",
                    fingerprint
                )
                
                if existing:
                    continue  # Skip duplicates
                
                # Insert new knowledge fact
                await db.execute_query(
                    """
                    INSERT INTO external_knowledge 
                    (entity_id, source_kg, entity_type, entity_name, relationship_type, 
                     related_entity_name, source_link, metadata, fingerprint)
                    VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
                    """,
                    fact.get("entity_id", ""),
                    fact.get("source_kg", "unknown"),
                    fact.get("entity_type", "unknown"),
                    fact.get("entity_name", ""),
                    fact.get("relationship_type"),
                    fact.get("related_entity_name"),
                    fact.get("source_link"),
                    fact.get("metadata", {}),
                    fingerprint
                )
                
                ingested_count += 1
            
            logger.info(f"Ingested {ingested_count} new knowledge facts")
            return ingested_count
            
        except Exception as e:
            logger.error(f"Knowledge ingestion failed: {e}")
            return 0
    
    def _extract_search_terms(self, query: str) -> List[str]:
        """Extract meaningful search terms from query."""
        # Remove common stop words and extract key terms
        stop_words = {'is', 'are', 'what', 'how', 'the', 'a', 'an', 'and', 'or', 'but', 'in', 'on', 'at', 'to', 'for', 'of', 'with'}
        
        # Simple tokenization and filtering
        words = re.findall(r'\b\w+\b', query.lower())
        search_terms = [word for word in words if word not in stop_words and len(word) > 2]
        
        return search_terms[:5]  # Limit to top 5 terms
    
    def _generate_fingerprint(self, fact: Dict[str, Any]) -> str:
        """Generate unique fingerprint for knowledge fact."""
        # Create fingerprint from key identifying fields
        fingerprint_data = f"{fact.get('entity_name', '')}-{fact.get('relationship_type', '')}-{fact.get('related_entity_name', '')}-{fact.get('source_kg', '')}"
        return hashlib.sha256(fingerprint_data.encode()).hexdigest()[:64]

class UserProfileManager:
    """Manages user profile creation and updates."""
    
    @staticmethod
    async def process_big_five_assessment(answers: List[float]) -> Dict[str, float]:
        """Process Big Five assessment answers into OCEAN scores."""
        if len(answers) != 10:
            raise ValueError("Big Five assessment requires exactly 10 answers")
        
        return {
            "openness": sum(answers[0:2]) / 2,
            "conscientiousness": sum(answers[2:4]) / 2,
            "extraversion": sum(answers[4:6]) / 2,
            "agreeableness": sum(answers[6:8]) / 2,
            "neuroticism": sum(answers[8:10]) / 2
        }
    
    @staticmethod
    async def create_profile_vector(
        test_scores: Dict[str, Dict[str, float]], 
        test_metadata: Dict[str, Dict[str, Any]]
    ) -> List[float]:
        """Create weighted profile vector from test scores."""
        weighted_vector = []
        
        for test_name, scores in test_scores.items():
            metadata = test_metadata.get(test_name, {})
            weight = metadata.get("validity_weight", 1.0)
            
            for score in scores.values():
                weighted_vector.append(score * weight)
        
        # Append initial engagement preference
        weighted_vector.append(0.0)
        
        return weighted_vector
    
    @staticmethod
    async def assign_cluster(profile_vector: List[float]) -> int:
        """Assign cluster based on profile vector using simple clustering."""
        # Simplified clustering for MVP - will be replaced with trained model
        vector_array = np.array(profile_vector[:-1])  # Exclude preference
        
        # Define prototype clusters (Big Five averages)
        cluster_prototypes = np.array([
            [3.0, 3.0, 3.0, 3.0, 3.0],  # Cluster 0: Average
            [4.0, 2.0, 4.0, 2.0, 4.0]   # Cluster 1: High O/E/N, Low C/A
        ])
        
        # Find closest cluster
        if len(vector_array) >= 5:
            distances = [np.linalg.norm(vector_array[:5] - prototype) 
                        for prototype in cluster_prototypes]
            return int(np.argmin(distances))
        
        return 0  # Default cluster

        assessments: List of test assessments with flexible score structure.

        Dict with status and user_id.

        HTTPException: If test_name is unknown or processing fails.
    """

    if not assessments:
        raise HTTPException(status_code=400, detail="No assessments provided")

    try:
        # Fetch test metadata for all assessments
        test_metadata = {}
        for assessment in assessments:
            meta = await db.fetch_one(
                "SELECT validity_weight FROM test_metadata WHERE test_name = $1",
                assessment.test_name
            )
            if not meta:
                raise HTTPException(
                    status_code=400, 
                    detail=f"Unknown test: {assessment.test_name}"
                )
            test_metadata[assessment.test_name] = meta
        
        # Build modular assessments structure and profile vector
        assessments_data = []
        weighted_vector = []
        
        for assessment in assessments:
            # Convert scores to dictionary format
            scores_dict = {score.score_name: score.value for score in assessment.scores}
            
            # Store assessment data
            assessments_data.append({
                "test_name": assessment.test_name,
                "scores": scores_dict,
                "confidence": assessment.confidence,
                "timestamp": pd.Timestamp.now().isoformat()
            })
            
            # Build weighted vector
            validity_weight = test_metadata[assessment.test_name]["validity_weight"]
            for score in assessment.scores:
                weighted_value = score.value * validity_weight * assessment.confidence
                weighted_vector.append(weighted_value)

        weighted_vector.append(0.0)

        cluster_id = await profile_manager.assign_cluster(weighted_vector)

               assessments = $2, profile_vector = $3, cluster_id = $4, updated_at = CURRENT_TIMESTAMP""",
            user_id, json.dumps(assessments_data), weighted_vector, cluster_id
        )
        
        log_info("User onboarded with modular assessments", user_id=user_id, 
                cluster_id=cluster_id, assessment_types=[a["test_name"] for a in assessments_data])

    except HTTPException:

"""
TPN-Prime 5.0 Safety Governor with NLI-based contradiction detection
"""
import logging
from typing import List, Dict
from transformers import pipeline
from shared.models import ContradictionAlert

logger = logging.getLogger(__name__)

class SafetyGovernor:
    """
    Safety service that uses Natural Language Inference (NLI) to detect
    contradictions between AI responses and verified facts.
    """
    
    def __init__(self):
        try:
            # Using a reliable NLI model for contradiction detection
            self.nli_pipeline = pipeline(
                "text-classification", 
                model="roberta-large-mnli",
                device=-1  # Use CPU for stability
            )
            self.model_loaded = True
            logger.info("Safety Governor NLI model loaded successfully")
        except Exception as e:
            logger.error(f"Failed to load NLI model: {e}")
            self.model_loaded = False
    
    def check_for_contradictions(self, draft_response: str, facts: List[Dict]) -> List[ContradictionAlert]:
        """
        Uses an NLI model to check for contradictions between the AI's response and verified facts.
        
        Args:
            draft_response: The AI's draft response text
            facts: List of verified facts from the Super-Graph
            
        Returns:
            List of detected contradictions with confidence scores
        """
        if not self.model_loaded:
            logger.warning("NLI model not available, skipping contradiction check")
            return []
        
        contradictions = []
        
        try:
            for fact in facts:
                # Create a clear premise from the structured fact
                premise = self._format_fact_as_premise(fact)
                hypothesis = draft_response
                
                # The NLI model expects a dictionary with text and text_pair
                nli_input = {"text": premise, "text_pair": hypothesis}
                
                # Get all scores from the NLI model
                results = self.nli_pipeline(nli_input, return_all_scores=True)
                
                # Check the results for a contradiction label with high confidence
                for result in results[0]:
                    if result['label'] == 'CONTRADICTION' and result['score'] > 0.8:
                        contradictions.append(ContradictionAlert(
                            contradicted_fact=premise,
                            ai_statement=hypothesis,
                            confidence=result['score']
                        ))
                        break
        
        except Exception as e:
            logger.error(f"Contradiction detection failed: {e}")
        
        if contradictions:
            logger.warning(f"Detected {len(contradictions)} contradictions in AI response")
        
        return contradictions
    
    def _format_fact_as_premise(self, fact: Dict) -> str:
        """Format a structured fact into a clear premise statement."""
        entity_name = fact.get('entity_name', '')
        relationship_type = fact.get('relationship_type', '')
        related_entity_name = fact.get('related_entity_name', '')
        
        if relationship_type and related_entity_name:
            return f"{entity_name} {relationship_type} {related_entity_name}."
        else:
            return f"{entity_name} is a known entity in medical literature."
    
    def validate_response_safety(self, draft_response: str, context: Dict) -> Dict[str, any]:
        """
        Comprehensive safety validation of AI response.
        
        Args:
            draft_response: AI's draft response
            context: Full context including facts and therapeutic components
            
        Returns:
            Validation result with safety status and any issues found
        """
        safety_result = {
            "is_safe": True,
            "contradictions": [],
            "warnings": [],
            "recommendations": []
        }
        
        try:
            # Check for contradictions with external facts
            factual_knowledge = context.get('factual_knowledge', [])
            if factual_knowledge:
                contradictions = self.check_for_contradictions(draft_response, factual_knowledge)
                safety_result["contradictions"] = [c.dict() for c in contradictions]
                
                if contradictions:
                    safety_result["is_safe"] = False
                    safety_result["recommendations"].append(
                        "Response contradicts verified medical facts and should be revised"
                    )
            
            # Additional safety checks
            safety_result["warnings"].extend(self._check_response_appropriateness(draft_response))
            
            return safety_result
            
        except Exception as e:
            logger.error(f"Safety validation failed: {e}")
            return {
                "is_safe": False,
                "contradictions": [],
                "warnings": ["Safety validation system error"],
                "recommendations": ["Manual review recommended"]
            }
    
    def _check_response_appropriateness(self, response: str) -> List[str]:
        """Additional appropriateness checks for therapeutic responses."""
        warnings = []
        response_lower = response.lower()
        
        # Check for inappropriate medical advice
        medical_red_flags = [
            'diagnose', 'diagnosis', 'prescribed', 'prescribe', 
            'stop taking', 'discontinue medication', 'medical emergency'
        ]
        
        for flag in medical_red_flags:
            if flag in response_lower:
                warnings.append(f"Response may contain inappropriate medical advice: '{flag}'")
        
        # Check for crisis indicators that need special handling
        crisis_indicators = ['suicide', 'self-harm', 'kill myself', 'end it all']
        
        for indicator in crisis_indicators:
            if indicator in response_lower:
                warnings.append("Response addresses crisis topics - ensure appropriate resources are provided")
        
        return warnings

"""
TPN-Prime 5.0 Discovery Analysis - Finding novel therapeutic pathways
"""
import asyncio
import pandas as pd

class DiscoveryEngine:
    """
    Offline analysis engine to discover novel therapeutic pathways
    from successful user interactions using Market Basket Analysis.
    """
    
    def __init__(self):
        self.min_support_count = 2  # Minimum occurrences to consider a pattern
        self.min_confidence = 0.6   # Minimum confidence for association rules
    
    async def analyze_discovery_data(self) -> Dict[str, any]:
        """
        Performs market basket analysis on successful interactions
        to find novel therapeutic component combinations.
        
        Returns:
            Dictionary with discovered patterns and insights
        """
        try:
            # Fetch successful interaction data
            interaction_data = await self._fetch_successful_interactions()
            
            if not interaction_data:
                logger.warning("No successful interaction data available for discovery")
                return {"patterns": [], "insights": [], "recommendations": []}
            
            # Convert to DataFrame for analysis
            df = pd.DataFrame(interaction_data)
            
            # Perform market basket analysis
            frequent_patterns = self._find_frequent_patterns(df)
            association_rules = self._generate_association_rules(df, frequent_patterns)
            
            # Generate insights
            insights = self._generate_insights(frequent_patterns, association_rules)
            
            # Create recommendations for new pathways
            recommendations = self._create_pathway_recommendations(frequent_patterns)
            
            log_info("Discovery analysis completed",

                    rules_generated=len(association_rules),
                    interactions_analyzed=len(df))
            
            return {
                "patterns": frequent_patterns,
                "association_rules": association_rules,
                "insights": insights,
                "recommendations": recommendations,
                "analysis_metadata": {
                    "interactions_count": len(df),
                    "min_support": self.min_support_count,
                    "min_confidence": self.min_confidence
                }
            }
            
        except Exception as e:
            logger.error(f"Discovery analysis failed: {e}")
            return {"error": str(e), "patterns": [], "insights": [], "recommendations": []}
    
    async def _fetch_successful_interactions(self) -> List[Dict]:
        """Fetch interaction data where user feedback was highly positive."""
        try:
            # Query for high-effectiveness training examples
            results = await db.fetch_all(
                """
                SELECT te.user_id, te.context_data, te.user_feedback_score, te.effectiveness_score
                FROM training_examples te
                WHERE te.user_feedback_score >= 4.0 
                AND te.effectiveness_score IS NOT NULL
                AND te.context_data IS NOT NULL
                AND te.is_validated = TRUE
                ORDER BY te.created_at DESC

                """)
            
            interaction_data = []
            for row in results:
                context = row["context_data"]
                if context and "therapeutic_components" in context:
                    components = [comp.get("component_id", "") for comp in context["therapeutic_components"]]
                    components = [c for c in components if c]  # Filter empty strings
                    
                    if components:
                        interaction_data.append({
                            "user_id": row["user_id"],
                            "components": components,
                            "feedback_score": row["user_feedback_score"],
                            "effectiveness_score": row["effectiveness_score"]
                        })
            
            # If no real data, use synthetic data for demonstration
            if not interaction_data:
                interaction_data = self._generate_synthetic_data()
            
            return interaction_data
            
        except Exception as e:
            logger.error(f"Failed to fetch successful interactions: {e}")
            return self._generate_synthetic_data()
    
    def _generate_synthetic_data(self) -> List[Dict]:

    def _find_frequent_patterns(self, df: pd.DataFrame) -> List[Dict]:
        """Find frequently occurring component combinations."""
        frequent_itemsets = defaultdict(int)
        
        for components_list in df['components']:
            # Consider combinations of 2 or more components
            for combo_size in range(2, len(components_list) + 1):
                for combo in combinations(sorted(components_list), combo_size):
                    frequent_itemsets[combo] += 1
        
        # Filter by minimum support count
        patterns = []
        for combo, count in frequent_itemsets.items():
            if count >= self.min_support_count:
                patterns.append({
                    "components": list(combo),
                    "support_count": count,
                    "support_ratio": count / len(df)
                })
        
        # Sort by support count descending
        patterns.sort(key=lambda x: x["support_count"], reverse=True)
        
        return patterns
    
    def _generate_association_rules(self, df: pd.DataFrame, frequent_patterns: List[Dict]) -> List[Dict]:
        """Generate association rules from frequent patterns."""
        rules = []
        
        for pattern in frequent_patterns:
            if len(pattern["components"]) < 2:
                continue
                
            components = pattern["components"]
            
            # Generate rules: if A then B
            for i in range(1, len(components)):
                for antecedent in combinations(components, i):
                    consequent = tuple(c for c in components if c not in antecedent)
                    
                    if not consequent:
                        continue
                    
                    # Calculate confidence: support(A ∪ B) / support(A)
                    antecedent_support = self._calculate_support(df, list(antecedent))
                    full_support = pattern["support_count"]
                    
                    if antecedent_support > 0:
                        confidence = full_support / antecedent_support
                        
                        if confidence >= self.min_confidence:
                            rules.append({
                                "antecedent": list(antecedent),
                                "consequent": list(consequent),
                                "confidence": confidence,
                                "support_count": full_support,
                                "lift": confidence / (self._calculate_support(df, list(consequent)) / len(df))
                            })
        
        # Sort by confidence descending
        rules.sort(key=lambda x: x["confidence"], reverse=True)
        
        return rules
    
    def _calculate_support(self, df: pd.DataFrame, components: List[str]) -> int:
        """Calculate support count for a set of components."""
        count = 0
        for components_list in df['components']:
            if all(comp in components_list for comp in components):
                count += 1
        return count
    
    def _generate_insights(self, patterns: List[Dict], rules: List[Dict]) -> List[str]:
        """Generate human-readable insights from the analysis."""
        insights = []
        
        if patterns:
            top_pattern = patterns[0]
            insights.append(
                f"Most frequent successful combination: {', '.join(top_pattern['components'])} "
                f"(appeared in {top_pattern['support_count']} successful cases)"
            )
        
        if rules:
            top_rule = rules[0]
            insights.append(
                f"Strongest association rule: When using {', '.join(top_rule['antecedent'])}, "
                f"also using {', '.join(top_rule['consequent'])} succeeds "
                f"{top_rule['confidence']:.1%} of the time"
            )
        
        # Component frequency analysis
        component_freq = defaultdict(int)
        for pattern in patterns:
            for comp in pattern["components"]:
                component_freq[comp] += pattern["support_count"]
        
        if component_freq:
            most_effective = max(component_freq.items(), key=lambda x: x[1])
            insights.append(
                f"Most effective component overall: {most_effective[0]} "
                f"(appeared in {most_effective[1]} successful combinations)"
            )
        
        return insights
    
    def _create_pathway_recommendations(self, patterns: List[Dict]) -> List[Dict]:
        """Create recommendations for new therapeutic pathways."""
        recommendations = []
        
        for pattern in patterns[:5]:  # Top 5 patterns
            if pattern["support_count"] >= self.min_support_count:
                recommendations.append({
                    "pathway_name": f"Discovery Pathway: {'-'.join(pattern['components'][:2])}",
                    "components": pattern["components"],
                    "evidence_strength": "emerging" if pattern["support_count"] < 5 else "moderate",
                    "success_rate": pattern["support_ratio"],
                    "recommendation": f"Consider creating a formal pathway combining these {len(pattern['components'])} components",
                    "next_steps": [
                        "Validate with clinical experts",
                        "Design controlled pilot study",
                        "Develop implementation protocols"
                    ]
                })
        
        return recommendations

class BayesianPathwayModel(nn.Module):
    """Simple Bayesian neural network for pathway recommendation."""
    
    def __init__(self, input_dim: int, hidden_dim: int = 64, output_dim: int = 2):
        super().__init__()
        self.fc1 = nn.Linear(input_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, hidden_dim)
        self.fc3 = nn.Linear(hidden_dim, output_dim)
        self.dropout = nn.Dropout(0.1)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = torch.relu(self.fc1(x))
        x = self.dropout(x)
        x = torch.relu(self.fc2(x))
        x = self.dropout(x)
        x = torch.sigmoid(self.fc3(x))
        return x

class PathwayGenerator:
    """Generates personalized therapeutic pathways using Bayesian learning."""
    
    def __init__(self):
        # Default pathway priors
        self.default_priors = {
            "pathway_cbt": norm(0.5, 0.1),
            "pathway_mindfulness": norm(0.3, 0.1),
            "pathway_exercise": norm(0.4, 0.1),
            "pathway_nutrition": norm(0.3, 0.1)
        }
        
        # Cluster-specific priors (learned from data)
        self.cluster_priors = {
            0: {  # Average cluster
                "pathway_cbt": norm(0.6, 0.1),
                "pathway_mindfulness": norm(0.4, 0.1),
                "pathway_exercise": norm(0.3, 0.1),
                "pathway_nutrition": norm(0.2, 0.1)
            },
            1: {  # High O/E/N, Low C/A cluster
                "pathway_cbt": norm(0.4, 0.1),
                "pathway_mindfulness": norm(0.6, 0.1),
                "pathway_exercise": norm(0.5, 0.1),
                "pathway_nutrition": norm(0.3, 0.1)
            }
        }
        
        # Cluster model (will be trained on actual data)
        self.cluster_model = KMeans(n_clusters=2, random_state=42)
        self.scaler = StandardScaler()
        
        # Initialize Bayesian model
        self.bayesian_model = BayesianPathwayModel(input_dim=6)
        
        # Reward weights for multi-objective optimization
        self.reward_weights = {
            "engagement": 0.33,
            "sentiment": 0.33,
            "insight": 0.33,
            "adherence": 0.01  # Future metric
        }
        
    async def initialize_cluster_model(self) -> None:
        """Initialize cluster model with existing user data."""
        try:
            profiles = await db.fetch_all(
                "SELECT profile_vector, cluster_id FROM user_profiles WHERE profile_vector IS NOT NULL"
            )
            
            if len(profiles) > 10:  # Minimum data for clustering
                vectors = np.array([p["profile_vector"][:-1] for p in profiles])  # Exclude preference
                vectors = self.scaler.fit_transform(vectors)
                self.cluster_model.fit(vectors)
                log_info("Cluster model initialized", sample_count=len(profiles))
            else:
                log_info("Insufficient data for cluster model training", sample_count=len(profiles))
                
        except Exception as e:
            logger.error(f"Failed to initialize cluster model: {e}")
    
    def calculate_reward(self, user_data: Dict) -> float:
        """
        Calculate multi-objective reward score.
        
        Args:
            user_data: Dictionary containing user interaction data.
            
        Returns:
            Float reward score between 0 and 1.
        """
        try:
            # Engagement component (normalized time spent)
            time_spent = user_data.get("engagement_metrics", {}).get("time_spent", 0)
            engagement = min(1.0, time_spent / 60.0)  # Normalize to minutes
            
            # Sentiment component (scale from [-1,1] to [0,1])
            sentiment = user_data.get("sentiment_score", 0)
            sentiment_normalized = (sentiment + 1) / 2
            
            # Insight component (count normalized)
            insight_count = sum(user_data.get("insight_markers", {}).values())
            insight_normalized = min(1.0, insight_count / 3.0)
            
            # Weighted combination
            reward = (
                self.reward_weights["engagement"] * engagement +
                self.reward_weights["sentiment"] * sentiment_normalized +
                self.reward_weights["insight"] * insight_normalized
            )
            
            return max(0.0, min(1.0, reward))
            
        except Exception as e:
            logger.error(f"Reward calculation failed: {e}")
            return 0.5  # Default neutral reward
    
    def predict_cluster(self, profile_vector: List[float]) -> int:
        """Predict cluster for user profile vector."""
        try:
            if len(profile_vector) < 5:
                return 0
            
            # Use only psychometric components (exclude preference)
            vector = np.array(profile_vector[:-1]).reshape(1, -1)
            vector = self.scaler.transform(vector)
            
            cluster = self.cluster_model.predict(vector)[0]
            return int(cluster)
            
        except Exception as e:
            logger.error(f"Cluster prediction failed: {e}")
            return 0  # Default cluster
    
    async def generate_pathway_scores(
        self, 
        user_id: str, 
        profile_vector: List[float], 
        user_data: Dict
    ) -> Tuple[Dict[str, float], int, float]:
        """
        Generate pathway scores using Bayesian inference.
        
        Args:
            user_id: Unique user identifier.
            profile_vector: User's dynamic profile vector.
            user_data: Recent user interaction data.
            
        Returns:
            Tuple of (pathway_scores, cluster_id, confidence).
        """
        try:
            # Predict cluster
            cluster_id = self.predict_cluster(profile_vector)
            
            # Get cluster-specific priors
            priors = self.cluster_priors.get(cluster_id, self.default_priors)
            
            # Calculate reward from user interactions
            reward = self.calculate_reward(user_data)
            
            # Create likelihood distribution around observed reward
            likelihood = norm(reward, 0.2)
            
            # Calculate posterior for each pathway using Bayesian inference
            pathways = {}
            for pathway_name, prior in priors.items():
                # P(pathway|data) ∝ P(data|pathway) * P(pathway)
                prior_score = prior.pdf(0.5)  # Prior at mean effectiveness
                likelihood_score = likelihood.pdf(0.5)  # Likelihood at target
                
                # Simple Bayesian update (normalized)
                posterior = prior_score * likelihood_score
                pathways[pathway_name] = float(posterior)
            
            # Normalize scores
            total_score = sum(pathways.values())
            if total_score > 0:
                pathways = {k: v / total_score for k, v in pathways.items()}
            
            # Calculate confidence based on session count and consistency
            session_count = user_data.get("session_count", 1)
            confidence = min(0.95, session_count / 100.0)
            
            return pathways, cluster_id, confidence
            
        except Exception as e:
            logger.error(f"Pathway generation failed: {e}")
            # Return default pathways
            return {
                "pathway_cbt": 0.4,
                "pathway_mindfulness": 0.3,
                "pathway_exercise": 0.2,
                "pathway_nutrition": 0.1
            }, 0, 0.5
    
    async def generate_pathway(
        self,
        user_id: str,
        profile_vector: List[float],
        user_data: Dict
    ) -> PathwayResponse:
        """
        Main pathway generation method.
        
        Args:
            user_id: Unique user identifier.
            profile_vector: User's dynamic profile vector.
            user_data: User interaction data.
            
        Returns:
            PathwayResponse with generated pathways and metadata.
        """
        with pathway_generation_duration.time():
            try:
                # Generate pathway scores
                pathways, cluster_id, confidence = await self.generate_pathway_scores(
                    user_id, profile_vector, user_data
                )
                
                # Calculate risk and complexity scores (simplified for MVP)
                risk_score = 0.5  # Default moderate risk
                complexity = 3    # Default moderate complexity
                
                # Generate unique pathway ID
                pathway_id = f"pathway_{user_id}_{uuid.uuid4().hex[:8]}"
                
                log_info("Pathway generated", user_id=user_id, pathway_id=pathway_id,
                        cluster_id=cluster_id, confidence=confidence)
                
                return PathwayResponse(
                    user_id=user_id,
                    pathway_id=pathway_id,
                    pathways=pathways,
                    cluster_id=cluster_id,
                    confidence=confidence,
                    risk_score=risk_score,
                    complexity=complexity
                )
                
            except Exception as e:
                logger.error(f"Pathway generation failed for user {user_id}: {e}")
                raise HTTPException(status_code=500, detail="Pathway generation failed")

class ValidationClassifier:

        }

    async def classify_pathway(self, pathway_data: Dict) -> int:
        """
        Classify pathway into validation tier (1-3).
        
        Args:
            pathway_data: Dictionary containing pathway information.
            
        Returns:
            Integer tier (1=auto, 2=AI-assisted, 3=full review).
        """
        try:
            # Extract features
            risk_score = pathway_data.get("risk_score", 0.5)
            complexity = pathway_data.get("complexity", 2)
            confidence = pathway_data.get("confidence", 0.5)
            user_count = pathway_data.get("user_count", 1)
            
            # Create feature vector
            features = np.array([[risk_score, complexity, confidence, user_count]])
            features_scaled = self.scaler.transform(features)
            
            # Predict tier
            tier = self.model.predict(features_scaled)[0]
            
            # Validate against rule-based criteria as backup
            if risk_score <= 0.3 and complexity <= 2:
                tier = min(tier, 1)
            elif risk_score >= 0.8 or complexity >= 5:
                tier = max(tier, 3)
            
            return int(tier)
            
        except Exception as e:
            logger.error(f"Pathway classification failed: {e}")
            return 2  # Default to AI-assisted review
    
    async def generate_summary(self, pathway_data: Dict) -> str:
        """
        Generate AI summary for pathway validation.
        
        Args:
            pathway_data: Dictionary containing pathway information.
            
        Returns:
            String summary of pathway for review.
        """
        try:
            if not self.summarizer:
                return f"Summary unavailable for {pathway_data.get('pathway_id', 'unknown')}"
            
            # Create text description
            pathway_id = pathway_data.get("pathway_id", "unknown")
            confidence = pathway_data.get("confidence", 0.5)
            user_count = pathway_data.get("user_count", 0)
            risk_score = pathway_data.get("risk_score", 0.5)
            pathways = pathway_data.get("pathway_data", {})
            
            text = f"""
            Pathway {pathway_id} shows confidence level of {confidence:.2f} based on {user_count} users.
            Risk assessment indicates {risk_score:.2f} risk level. 
            Recommended interventions include: {', '.join(pathways.keys() if pathways else ['none'])}.
            Clinical effectiveness varies by user cluster and individual response patterns.
            Requires validation for safety and efficacy before deployment.
            """
            
            # Generate summary
            summary = self.summarizer(
                text.strip(),
                max_length=50,
                min_length=20,
                do_sample=False
            )[0]['summary_text']
            
            return summary
            
        except Exception as e:
            logger.error(f"Summary generation failed: {e}")
            return f"Summary generation failed for pathway {pathway_data.get('pathway_id', 'unknown')}"
    
    async def calculate_priority_score(self, pathway_data: Dict) -> float:
        """
        Calculate priority score for Tier 3 review queue.
        
        Args:
            pathway_data: Dictionary containing pathway information.
            
        Returns:
            Float priority score (higher = more urgent).
        """
        try:
            confidence = pathway_data.get("confidence", 0.5)
            user_count = pathway_data.get("user_count", 1)
            risk_score = pathway_data.get("risk_score", 0.5)
            
            # Priority = confidence * user_impact - risk_penalty
            user_impact = min(user_count / 10.0, 1.0)  # Normalize to [0,1]
            risk_penalty = risk_score * 0.5
            
            priority = confidence * user_impact - risk_penalty
            return max(0.0, min(1.0, priority))
            
        except Exception as e:
            logger.error(f"Priority calculation failed: {e}")
            return 0.5
    
    async def prioritize_reviews(self, pathways: List[Dict]) -> List[Dict]:
        """
        Sort pathways by priority for review queue.
        
        Args:
            pathways: List of pathway dictionaries.
            
        Returns:
            List of pathways sorted by priority (highest first).
        """
        try:
            # Calculate priority scores
            for pathway in pathways:
                pathway["priority_score"] = await self.calculate_priority_score(pathway)
            
            # Sort by priority
            return sorted(pathways, key=lambda p: p.get("priority_score", 0), reverse=True)
            
        except Exception as e:
            logger.error(f"Review prioritization failed: {e}")
            return pathways

    """
    Validate pathway and assign appropriate tier.
    
    Args:
        request: ValidationRequest with pathway data.
        
    Returns:
        ValidationResponse with tier assignment and metadata.

        HTTPException: If validation fails.
    """
    request_counter.labels(endpoint="validate").inc()
    
    with validation_duration.time():
        try:
            # Classify pathway into tier
            tier = await validation_classifier.classify_pathway(request.dict())
            
            # Generate summary for Tier 2+ pathways
            summary = None
            if tier >= 2:
                summary = await validation_classifier.generate_summary(request.dict())
            
            # Calculate priority score for Tier 3 pathways
            priority_score = None
            if tier == 3:
                priority_score = await validation_classifier.calculate_priority_score(
                    request.dict()
                )
            
            # Determine status
            status = "VALIDATED" if tier == 1 else "PENDING"
            
            # Update pathway in database
            await db.execute_query(
                """UPDATE pathways 
                   SET validation_tier = $1, status = $2, updated_at = CURRENT_TIMESTAMP
                   WHERE pathway_id = $3""",
                tier, status, request.pathway_id
            )
            
            # Store validation review if needed
            if tier >= 2:
                await db.execute_query(
                    """INSERT INTO validation_reviews 
                       (pathway_id, review_status, review_notes, priority_score)
                       VALUES ($1, $2, $3, $4)""",
                    request.pathway_id, "PENDING", summary, priority_score
                )
            
            log_info("Pathway validated", pathway_id=request.pathway_id,
                    tier=tier, status=status, confidence=request.confidence)

                pathway_id=request.pathway_id,
                tier=tier,

            )

            raise HTTPException(status_code=500, detail="Pathway validation failed")

@app.get("/review/queue", response_model=List[Dict])
async def get_review_queue(tier: int = None, limit: int = 20) -> List[Dict]:
    """
    Get pathways pending review, optionally filtered by tier.
    
    Args:
        tier: Optional tier filter (2 or 3).
        limit: Maximum number of pathways to return.
        
    Returns:
        List of pathways pending review, sorted by priority.
    """
    request_counter.labels(endpoint="review_queue").inc()
    
    try:
        # Build query
        where_clause = "WHERE p.status = 'PENDING'"
        params = []
        
        if tier:
            where_clause += " AND p.validation_tier = $1"
            params.append(tier)
            limit_param = "$2"
        else:
            limit_param = "$1"
        params.append(limit)
        
        query = f"""
            SELECT p.pathway_id, p.cluster_id, p.pathway_data, p.confidence,
                   p.risk_score, p.complexity, p.validation_tier, p.created_at,
                   vr.review_notes, vr.priority_score
            FROM pathways p
            LEFT JOIN validation_reviews vr ON p.pathway_id = vr.pathway_id
            {where_clause}
            ORDER BY vr.priority_score DESC NULLS LAST, p.created_at ASC
            LIMIT {limit_param}
        """
        
        pathways = await db.fetch_all(query, *params)
        
        # Format response
        result = []
        for p in pathways:
            result.append({
                "pathway_id": p["pathway_id"],
                "cluster_id": p["cluster_id"],
                "pathway_data": p["pathway_data"],
                "confidence": p["confidence"],
                "risk_score": p["risk_score"],
                "complexity": p["complexity"],
                "validation_tier": p["validation_tier"],
                "summary": p["review_notes"],
                "priority_score": p["priority_score"],
                "created_at": p["created_at"].isoformat()
            })
        
        return result
        
    except Exception as e:
        logger.error(f"Failed to retrieve review queue: {e}")
        raise HTTPException(status_code=500, detail="Review queue retrieval failed")

class SignalReconciler:
    """Reconciles real-world evidence with clinical standards using confidence weighting."""
    
    def __init__(self):
        # Clinical standards database (evidence-based effectiveness scores)
        self.clinical_standards = {
            "pathway_cbt": {
                "effectiveness": 0.75,
                "evidence_level": "high",
                "studies": 150,
                "description": "Cognitive Behavioral Therapy - Strong evidence base"
            },
            "pathway_mindfulness": {
                "effectiveness": 0.65,
                "evidence_level": "moderate",
                "studies": 85,
                "description": "Mindfulness-based interventions - Moderate evidence"
            },
            "pathway_exercise": {
                "effectiveness": 0.70,
                "evidence_level": "high",
                "studies": 120,
                "description": "Exercise therapy - Strong evidence for ADHD"
            },
            "pathway_nutrition": {
                "effectiveness": 0.45,
                "evidence_level": "low",
                "studies": 25,
                "description": "Nutritional interventions - Limited evidence"
            }
        }
        
        # Reconciliation parameters
        self.base_real_world_weight = 0.7  # Base weight for real-world evidence
        self.confidence_scaling = 1.0      # Scaling factor for confidence
        self.evidence_level_weights = {
            "high": 1.0,
            "moderate": 0.8,
            "low": 0.6
        }
    
    async def get_clinical_score(self, pathway_name: str) -> Dict[str, float]:
        """
        Get clinical standard score for a pathway.
        
        Args:
            pathway_name: Name of the therapeutic pathway.
            
        Returns:
            Dict with clinical effectiveness and metadata.
        """
        standard = self.clinical_standards.get(pathway_name, {
            "effectiveness": 0.5,
            "evidence_level": "low",
            "studies": 0,
            "description": "Unknown pathway - default scoring"
        })
        
        return {
            "clinical_score": standard["effectiveness"],
            "evidence_weight": self.evidence_level_weights[standard["evidence_level"]],
            "study_count": standard["studies"],
            "description": standard["description"]
        }
    
    async def calculate_confidence_weights(
        self, 
        confidence: float, 
        evidence_level: str = "moderate"
    ) -> Dict[str, float]:
        """
        Calculate confidence-aware weights for reconciliation.
        
        Args:
            confidence: Real-world evidence confidence (0-1).
            evidence_level: Clinical evidence level (high/moderate/low).
            
        Returns:
            Dict with reconciliation weights.
        """
        try:
            # Real-world evidence weight scaled by confidence
            real_world_weight = self.base_real_world_weight * confidence
            
            # Clinical weight is complement, scaled by evidence level
            evidence_scale = self.evidence_level_weights.get(evidence_level, 0.8)
            clinical_weight = (1.0 - real_world_weight) * evidence_scale
            
            # Normalize weights to sum to 1
            total_weight = real_world_weight + clinical_weight
            if total_weight > 0:
                real_world_weight /= total_weight
                clinical_weight /= total_weight
            else:
                real_world_weight = 0.5
                clinical_weight = 0.5
            
            return {
                "real_world_weight": real_world_weight,
                "clinical_weight": clinical_weight,
                "confidence_factor": confidence,
                "evidence_factor": evidence_scale
            }
            
        except Exception as e:
            logger.error(f"Weight calculation failed: {e}")
            return {
                "real_world_weight": 0.5,
                "clinical_weight": 0.5,
                "confidence_factor": confidence,
                "evidence_factor": 0.8
            }
    
    async def reconcile_pathway_scores(
        self,
        pathway_scores: Dict[str, float],
        confidence: float,
        pathway_id: str
    ) -> Dict[str, any]:
        """
        Reconcile real-world pathway scores with clinical standards.
        
        Args:
            pathway_scores: Real-world effectiveness scores by pathway.
            confidence: Confidence in real-world evidence.
            pathway_id: Unique pathway identifier.
            
        Returns:
            Dict with reconciled scores and metadata.
        """
        try:
            reconciled_scores = {}
            reconciliation_details = {}
            
            for pathway_name, real_world_score in pathway_scores.items():
                # Get clinical standard
                clinical_data = await self.get_clinical_score(pathway_name)
                clinical_score = clinical_data["clinical_score"]
                evidence_level = "moderate"  # Default for MVP
                
                # Calculate weights
                weights = await self.calculate_confidence_weights(
                    confidence, evidence_level
                )
                
                # Reconcile scores: weighted average
                meta_score = (
                    weights["real_world_weight"] * real_world_score +
                    weights["clinical_weight"] * clinical_score
                )
                
                reconciled_scores[pathway_name] = meta_score
                reconciliation_details[pathway_name] = {
                    "real_world_score": real_world_score,
                    "clinical_score": clinical_score,
                    "meta_score": meta_score,
                    "weights": weights,
                    "clinical_metadata": clinical_data
                }
            
            # Calculate overall meta-score (weighted average of all pathways)
            if reconciled_scores:
                overall_meta_score = np.mean(list(reconciled_scores.values()))
            else:
                overall_meta_score = 0.5
            
            return {
                "pathway_id": pathway_id,
                "overall_meta_score": overall_meta_score,
                "pathway_scores": reconciled_scores,
                "reconciliation_details": reconciliation_details,
                "confidence": confidence
            }
            
        except Exception as e:
            logger.error(f"Reconciliation failed for pathway {pathway_id}: {e}")
            raise

async def reconcile_pathway(request: ReconciliationRequest) -> ReconciliationResponse:
    """
    Reconcile pathway scores with clinical standards.
    
    Args:
        request: ReconciliationRequest with pathway data.
        
    Returns:

    Raises:
        HTTPException: If reconciliation fails.

    with reconciliation_duration.time():

            # Perform reconciliation

            )
            
            # Update pathway in database
            await db.execute_query(

                   SET meta_score = $1, status = 'VALIDATED', updated_at = CURRENT_TIMESTAMP
                   WHERE pathway_id = $2""",
                reconciliation_result["overall_meta_score"],
                request.pathway_id
            )
            
            log_info("Pathway reconciled", 
                    pathway_id=request.pathway_id,
                    meta_score=reconciliation_result["overall_meta_score"],
                    confidence=request.confidence)
            
            return ReconciliationResponse(
                pathway_id=request.pathway_id,
                meta_score=reconciliation_result["overall_meta_score"],
                confidence=request.confidence,
                reconciliation_weights={
                    "real_world_weight": 0.7 * request.confidence,
                    "clinical_weight": 1.0 - (0.7 * request.confidence)
                }
            )
            
        except Exception as e:
            logger.error(f"Reconciliation failed for {request.pathway_id}: {e}")
            raise HTTPException(status_code=500, detail="Reconciliation failed")
