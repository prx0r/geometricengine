class TestEmbedder:
    def __init__(self):
        self.tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')
        self.model = BertModel.from_pretrained('bert-base-uncased')

    def embed(self, test_scores: Dict) -> torch.Tensor:
        text = " ".join([f"{trait} {score}" for test, traits in test_scores.items() for trait, score in traits.items()])
        inputs = self.tokenizer(text, return_tensors='pt', padding=True, truncation=True)
        outputs = self.model(**inputs)
        return outputs.last_hidden_state.mean(dim=1)

class ConversationTCN(torch.nn.Module):
    def __init__(self, input_dim=768, hidden_dim=256, num_layers=3):
        super().__init__()
        self.tcn = TemporalConvNet(input_dim, [hidden_dim] * num_layers, kernel_size=3)
        self.fc = torch.nn.Linear(hidden_dim, 256)

    def forward(self, history: torch.Tensor) -> torch.Tensor:
        x = self.tcn(history)
        return self.fc(x[-1])

class ResponsePredictor(torch.nn.Module):
    def __init__(self, in_channels=256, hidden_channels=128):
        super().__init__()
        self.conv1 = SAGEConv(in_channels, hidden_channels)
        self.conv2 = SAGEConv(hidden_channels, 1)

    def forward(self, x, edge_index):
        x = self.conv1(x, edge_index).relu()
        x = self.conv2(x, edge_index)
        return x

def modulate_vector(profile: Dict, context_vector: torch.Tensor, feedback_score: float) -> Dict:
    """Update DMPVN profile vector based on context and feedback."""
    delta = context_vector * feedback_score * 0.1
    for key in profile:
        profile[key] += delta[0, 0].item()
    return profile

def store_feedback(user_id: str, feedback: Feedback):
    """Store feedback in Super-Graph and Redis."""
    db.execute("""
        MATCH (u:User {id: $user_id})
        CREATE (r:ResponseSelection {option_id: $option_id, rating: $rating, comment: $comment, timestamp: $timestamp})
        CREATE (u)-[:SELECTED]->(r)
        CREATE (r)-[:FOR_COMPONENT]->(c:TherapeuticComponent {id: $component_id})
    """, {
        "user_id": user_id,
        "option_id": feedback.option_id,
        "rating": feedback.rating,
        "comment": feedback.comment,
        "timestamp": datetime.now().isoformat(),
        "component_id": feedback.component_id
    })
    redis_client.set(f"feedback:{user_id}:{feedback.component_id}", json.dumps(feedback.dict()))

def discover_therapies(interactions: List[Dict]) -> List[Dict]:
    """Discover new therapy combinations using Apriori."""
    df = pd.DataFrame([{t: 1 for t in i["therapies"]} for i in interactions]).fillna(0)
    frequent = apriori(df, min_support=0.1, use_colnames=True)
    rules = association_rules(frequent, metric="lift", min_threshold=1.2)
    return rules.to_dict('records')

@app.post("/process_input")
async def process_input(input: UserInput):
    """Process user input, detect juncture, and return response options."""
    juncture = detect_juncture(input.text, input.history)
    if not juncture:
        raise HTTPException(status_code=404, detail="No juncture detected")
    options = [opt for opt in juncture["learning_options"] if opt.get("safety_flag") != "REMOVE_FROM_TRAINING"]
    return {
        "component_id": juncture["component_id"],
        "juncture": juncture["source_validation"]["dialogue_example"]["client_says"],
        "options": options
    }

@app.post("/submit_feedback")
async def submit_feedback(feedback: Feedback):
    """Capture tester feedback and update DMPVN."""
    store_feedback(feedback.user_id, feedback)
    profile = json.loads(redis_client.get(f"profile:{feedback.user_id}").decode('utf-8') or "{}")
    history = torch.randn(30, 1, 768)  # Mock history
    context_vector = tcn(history)
    profile = modulate_vector(profile, context_vector, feedback.rating / 5.0)
    redis_client.set(f"profile:{feedback.user_id}", json.dumps(profile))
    return {"status": "Feedback processed"}
